activity <- read.csv("activity.csv")
activity<-activity %>% mutate(date=as.Date(activity$date,format="%Y-%m-%d"))
activityByDay<-activity %>%
    na.omit() %>%
    group_by(date) %>% 
    summarise(totalSteps=sum(steps))
